26 /*given string splits into token using given delimiter*/
27 char** str_split(char* a_str, const char a_delim)
28 {
29  char** result = 0;
30  size_t count = 0;
31  char* tmp = a_str;
32  char* last_comma = 0;
33  char delim[2];
34  delim[0] = a_delim;
35  delim[1] = 0;
36  /* Count how many elements will be extracted. */
37  while (*tmp)
38  {
39  if (a_delim == *tmp)
40  {
41  count++;
42  last_comma = tmp;
43  }
44  tmp++;
45  }
46  /* Add space for trailing token. */
47  count += last_comma < (a_str + strlen(a_str) - 1);
48  /* Add space for terminating null string so caller
49  knows where the list of returned strings ends. */
50  count++;
51  /*allocate memory for final result*/
52  result = malloc(sizeof(char*) * count);
53  if (result)
54  {
55  size_t idx = 0;
56  /*get a token one by one*/
57  char* token = strtok(a_str, delim);
58  while (token)
59  {
60  assert(idx < count);
61  *(result + idx++) = strdup(token);
62  token = strtok(0, delim);
63  }
64  assert(idx == count - 1);
65  *(result + idx) = 0;
66  }
67  return result;
68 }

208 /*remove leading and trailing white space from string*/
209 char *str_trim(char *str)
210 {
211  size_t len = 0;
212  char *frontp = str;
213  char *endp = NULL;
214  if( str == NULL ) { return NULL; }
215  if( str[0] == '\0' ) { return str; }
216  len = strlen(str);
217  endp = str + len;
218  /* Move the front and back pointers to address the first non-whitespace
219  * characters from each end.
220  */
221  while( isspace(*frontp) ) { ++frontp; }
222  if( endp != frontp )
223  {
224  while( isspace(*(--endp)) && endp != frontp ) {}
225  }
226  if( str + len - 1 != endp )
227  *(endp + 1) = '\0';
228  else if( frontp != str && endp == frontp )
229  *str = '\0';
230  /* Shift the string so that it starts at str so that if it's dynamically
231  * allocated, we can still free it on the returned pointer. Note the reuse
232  * of end to mean the front of the string buffer now.
233  */
234  endp = str;
235  if( frontp != str )
236  {
237  while( *frontp ) { *endp++ = *frontp++; }
238  *endp = '\0';
239  }
240  return str;
241 }
